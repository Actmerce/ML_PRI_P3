""" Gaussian Jordan  Ax = b.
    
        A:

        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""

def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    def turn_the_rest_to_zero(C, nr, rows):
        # 0
        for i in range(rows):
            if i == nr:
                pass
            else:
                scale = -C[i][nr]
                # r1 + r2*scale
                addScaledRow(C, i, nr, scale)

    r, c = shape(A)
    rb, cb = shape(b)

    # 
    if r != rb:
        return None
    
    M = augmentMatrix(A, b)

    matxRound(M, decPts)
    for i in range(r):  # i

        # 
        num = i
        while abs(M[i][i]) < epsilon:
            # 00
            num += 1

            if num > r-1:
                # 
                return None
            else:
                swapRows(M, i, num)

        scale = 1 / M[i][i]

        # 
        scaleRow(M, i, scale)

        # 
        turn_the_rest_to_zero(M, i, r)
    
    # b
    M = [row[-cb:] for row in M]
    matxRound(M, decPts)
    return M

def linearRegression(X,Y):
    # X,Y
    M_X = [[x, 1] for x in X]
    M_Y = [[y] for y in Y]
    
    # X^T
    t_X = transpose(M_X)
    
    # X^T * X
    a = matxMultiply(t_X, M_X)
    
    # X^T * Y
    b = matxMultiply(t_X, M_Y)
    
    # a 
    a_ = gj_Solve(a, [[1, 0], [0, 1]])
    
    h = matxMultiply(a_, b)
    matxRound(h, 4)
